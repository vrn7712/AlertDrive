void setup() {

Serial.begin(115200);

mpu.initialize();

}


void loop() {

// 1. Acquire 6-axis Motion Data

mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);


// 2. Calculate Pitch & Roll (Complementary Filter)

float pitch = atan(-ax / sqrt(pow(ay, 2) + pow(az, 2)));


// 3. Safety Threshold Check (35 degrees)

if (abs(pitch) > 35.0) {

digitalWrite(buzzer, HIGH);

sendTelemetry("CRITICAL_TILT");

}

}
